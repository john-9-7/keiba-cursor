                // フォーム送信を行う関数
                function submitFormOnChange() {
                    // 選択された日付と競馬場を取得
                    const date = document.getElementById('date_select').value;
                    const venueSelect = document.getElementById('venue_select');

                    // 現在のレース番号を取得（表示中のレース番号）
                    const currentRaceNumber = 1;

                    // 日付が変更された場合、競馬場の選択肢を更新
                    if (window.allRaces && window.allRaces[date]) {
                        // 現在選択されている競馬場を保持
                        const currentVenue = venueSelect.value;

                        // 変更された日付に存在する競馬場のリスト
                        const venues = Object.keys(window.allRaces[date]);

                        // 競馬場リストが空でない場合のみ処理
                        if (venues.length > 0) {
                            // 選択肢をクリア
                            venueSelect.innerHTML = '';

                            // 変更後の日付に現在の競馬場が存在するかどうか
                            let matchingVenueFound = false;

                            // 新しい選択肢を追加
                            venues.forEach(function(venue) {
                                const option = document.createElement('option');
                                option.value = venue;
                                option.textContent = venue;
                                venueSelect.appendChild(option);

                                // 現在選択されている競馬場が、変更後の日付にも存在するか
                                if (venue === currentVenue) {
                                    option.selected = true;
                                    matchingVenueFound = true;
                                }
                            });

                            // 現在の競馬場が変更後の日付に存在しない場合は最初の競馬場を選択
                            if (!matchingVenueFound) {
                                venueSelect.value = venues[0];
                            }

                            // 使用する競馬場を取得
                            const selectedVenue = venueSelect.value;

                            // 選択された競馬場のレースIDを取得
                            if (window.allRaces[date][selectedVenue] && window.allRaces[date][selectedVenue].length > 0) {
                                const races = window.allRaces[date][selectedVenue];

                                // レースを番号順にソート
                                races.sort((a, b) => parseInt(a.number) - parseInt(b.number));

                                // 同じレース番号のレースを探す
                                let targetRaceId = races[0].id; // デフォルトは最初のレース

                                for (let i = 0; i < races.length; i++) {
                                    if (parseInt(races[i].number) === currentRaceNumber) {
                                        // 同じレース番号が見つかったらそのIDを使用
                                        targetRaceId = races[i].id;
                                        break;
                                    }
                                }

                                // hidden_race_idを更新
                                document.getElementById('hidden_race_id').value = targetRaceId;

                                // 現在のタブ状態を隠し項目として追加
                                const form = document.getElementById('raceSelectForm');
                                const activeTab = document.querySelector('.tab-btn.active');
                                const currentTabType = activeTab ? activeTab.getAttribute('data-tab') : 'web';

                                // 既存の隠し項目があれば削除
                                const existingTabInput = form.querySelector('input[name="active_tab"]');
                                if (existingTabInput) existingTabInput.remove();

                                // 新しい隠し項目を追加
                                const tabStateInput = document.createElement('input');
                                tabStateInput.type = 'hidden';
                                tabStateInput.name = 'active_tab';
                                tabStateInput.value = currentTabType;
                                form.appendChild(tabStateInput);

                                // フォームを送信
                                document.getElementById('raceSelectForm').submit();
                            }
                        }
                    }
                }

                // グローバルスコープに馬印モーダル表示関数を定義
                function showMarkModal(element, horseId) {
                    const markSelectorModal = document.getElementById('markSelectorModal');
                    if (!markSelectorModal) {
                        console.error('馬印モーダルが見つかりません');
                        return;
                    }

                    // モーダルの位置を設定
                    const rect = element.getBoundingClientRect();

                    // ビューポート座標をそのまま使用（スクロール位置は既に含まれている）
                    let topPosition = rect.top; // 印欄と同じ高さに配置
                    let leftPosition = rect.right + 5; // 印欄の右側に5pxの余白を設けて配置

                    // モーダルのサイズを取得
                    const modalWidth = 180; // CSSで設定されている幅
                    const modalHeight = 120; // 推定の高さ

                    // 画面のサイズを取得
                    const windowWidth = window.innerWidth;
                    const windowHeight = window.innerHeight

                    // 右端がはみ出る場合は左に表示
                    if (leftPosition + modalWidth > windowWidth) {
                        leftPosition = rect.left - modalWidth - 5;
                    }

                    // 下端がはみ出る場合は上に調整
                    if (topPosition + modalHeight > windowHeight) {
                        topPosition = windowHeight - modalHeight - 10;
                    }

                    // 左端がはみ出る場合は右に表示
                    if (leftPosition < 10) {
                        leftPosition = rect.right + 5;
                    }

                    // 上端がはみ出る場合は下に調整
                    if (topPosition < 10) {
                        topPosition = 10;
                    }

                    markSelectorModal.style.top = topPosition + 'px';
                    markSelectorModal.style.left = leftPosition + 'px';

                    // モーダルを表示
                    markSelectorModal.style.display = 'block';
                    markSelectorModal.setAttribute('data-current-horse-id', horseId);
                }

                // 次の馬印に変更する関数
                function cycleNextMark(element, horseId) {
                    // モーダルが表示されていたら何もしない
                    const markSelectorModal = document.getElementById('markSelectorModal');
                    if (markSelectorModal && markSelectorModal.style.display === 'block') {

                // Cookie操作のヘルパー関数
                function setCookie(name, value, days) {
                    let expires = '';
                    if (days) {
                        const date = new Date();
                        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
                        expires = '; expires=' + date.toUTCString();
                    }
                    document.cookie = name + '=' + encodeURIComponent(value) + expires + '; path=/';
                }

                function getCookie(name) {
                    const nameEQ = name + '=';
                    const ca = document.cookie.split(';');
                    for (let i = 0; i < ca.length; i++) {
                        let c = ca[i];
                        while (c.charAt(0) === ' ') c = c.substring(1, c.length);
                        if (c.indexOf(nameEQ) === 0) return decodeURIComponent(c.substring(nameEQ.length, c
                            .length));
                    }
                    return null;
                }

                // ここから既存の機能を関数に分割
                function initHorseTableSort() {
                    const horseTable = document.getElementById('race-horses-table');
                    if (!horseTable) return;

                    const horseHeaders = horseTable.querySelectorAll('th.sortable');
                    const horseTbody = horseTable.querySelector('tbody');
                    const horseRows = Array.from(horseTbody.querySelectorAll('tr'));

                    // 全馬に着順が存在するかチェック
                    let allHaveFinishPosition = true;
                    const finishPositionCells = horseTable.querySelectorAll('tbody tr td.finish-position-col');
                    finishPositionCells.forEach(cell => {
                        const value = parseFloat(cell.getAttribute('data-value'));
                        // 999や未数値が1つでもあれば「全てに着順がある」条件を満たさない
                        if (isNaN(value) || value === 999) {
                            allHaveFinishPosition = false;
                        }
                    });

                    // 着順が全馬にない場合の単勝オッズ存在チェック
                    let hasValidWinOdds = false;
                    if (!allHaveFinishPosition) {
                        const winOddsCells = horseTable.querySelectorAll('tbody tr td.win-odds-col');
                        winOddsCells.forEach(cell => {
                            const value = parseFloat(cell.getAttribute('data-value'));
                            // 0でない有効な値がある場合
                            if (!isNaN(value) && value > 0) {
                                hasValidWinOdds = true;
                            }
                        });
                    }

                    // 現在のソート状態（デフォルトを着順・オッズの有無で変更）
                    let horseCurrentSort = {
                        column: allHaveFinishPosition ? 'finish-position' : (hasValidWinOdds ? 'win-odds' : 'bb-point'),
                        direction: allHaveFinishPosition ? 'asc' : (hasValidWinOdds ? 'asc' : 'desc')
                    };

                    // 着順が全馬に存在する場合、初期ソートを適用
                    if (allHaveFinishPosition) {
                        // 着順ヘッダーにソートアイコンを表示
                        const finishPositionHeader = horseTable.querySelector('th[data-sort="finish-position"]');
                        if (finishPositionHeader) {
                            const sortIcon = finishPositionHeader.querySelector('.sort-icon');
                            if (sortIcon) {
                                sortIcon.classList.add('asc');
                            }
                        }
                        // 初期ソートを実行
                        sortHorseTable('finish-position', 'asc');
                    } else if (hasValidWinOdds) {
                        // 着順が全馬にない場合で単勝オッズがある場合は単勝オッズの小さい順でソート
                        const winOddsHeader = horseTable.querySelector('th[data-sort="win-odds"]');
                        if (winOddsHeader) {
                            const sortIcon = winOddsHeader.querySelector('.sort-icon');
                            if (sortIcon) {
                                sortIcon.classList.add('asc');
                            }
                        }
                        sortHorseTable('win-odds', 'asc');
                    } else {
                        // どちらもない場合はBB指数の大きい順
                        const bbHeader = horseTable.querySelector('th[data-sort="bb-point"]');
                        if (bbHeader) {
                            const sortIcon = bbHeader.querySelector('.sort-icon');
                            if (sortIcon) {
                                sortIcon.classList.add('desc');
                            }
                        }
                        sortHorseTable('bb-point', 'desc');
                    }

                    // 推定人気順テーブルのソート関数（initHorseTableSort 内に定義）
                    function sortHorseTable(column, direction) {
                        const columnClassMap = {
                            'finish-position': 'finish-position-col',
                            'horse-number': 'horse-number-col',
                            'horse-mark': 'horse-mark-col',
                            'bb-point': 'bb-point-col',
                            'ability': 'ability-col',
                            'total-ability': 'total-ability-col',
                            'compilation-rank': 'compilation-rank-col',
                            'win-odds': 'win-odds-col',
                            'place-odds': 'place-odds-col',
                            'win-rate': 'win-rate-col',
                            'rentai-rate': 'rentai-rate-col',
                            'fukusho-rate': 'fukusho-rate-col',
                            'win-recovery': 'win-recovery-col',
                            'place-recovery': 'place-recovery-col'
                        };
                        const columnClass = columnClassMap[column];
                        if (!columnClass) return;

                        const sortedRows = horseRows.sort((a, b) => {
                            const cellA = a.querySelector(`td.${columnClass}`);
                            const cellB = b.querySelector(`td.${columnClass}`);

                            if (!cellA || !cellB) return 0;

                            // 馬印の特別処理
                            if (column === 'horse-mark') {
                                const markOrder = {
                                    '◎': 1,
                                    '○': 2,
                                    '▲': 3,
                                    '△': 4,
                                    '☆': 5,
                                    '✓': 6,
                                    '--': 7
                                };
                                const spanA = cellA.querySelector('span');
                                const spanB = cellB.querySelector('span');
                                const markA = spanA ? spanA.textContent : '';
                                const markB = spanB ? spanB.textContent : '';

                                // 「消」は常に最下位
                                if (markA === '消' && markB === '消') return 0;
                                if (markA === '消') return 1;
                                if (markB === '消') return -1;

                                const vA = markOrder[markA] || 9;
                                const vB = markOrder[markB] || 9;
                                return direction === 'asc' ? (vA - vB) : (vB - vA);
                            }

                            // 能力/総合の特別処理（SSS > SS > S > AA > A > B > C > D > E > F > G）
                            if (column === 'ability' || column === 'total-ability') {
                                const gradeOrder = {
                                    'SSS': 11,
                                    'SS': 10,
                                    'S': 9,
                                    'AA': 8,
                                    'A': 7,
                                    'B': 6,
                                    'C': 5,
                                    'D': 4,
                                    'E': 3,
                                    'F': 2,
                                    'G': 1
                                };
                                const rawA = (cellA.getAttribute('data-value') || '').toUpperCase().trim();
                                const rawB = (cellB.getAttribute('data-value') || '').toUpperCase().trim();
                                const isUnknownA = !(rawA in gradeOrder);
                                const isUnknownB = !(rawB in gradeOrder);
                                if (isUnknownA !== isUnknownB) {
                                    return isUnknownA ? 1 : -1; // 未知は常に最後
                                }
                                const vA = gradeOrder[rawA] || 0;
                                const vB = gradeOrder[rawB] || 0;
                                return direction === 'asc' ? (vA - vB) : (vB - vA);
                            }

                            // 単勝/複勝ソート時は、取消・除外・中止・失格（finish_position < 0）とオッズ0.0を常に最後へ
                            if (column === 'win-odds' || column === 'place-odds') {
                                const finCellA = a.querySelector('td.finish-position-col');
                                const finCellB = b.querySelector('td.finish-position-col');
                                const finValA = finCellA ? parseFloat(finCellA.getAttribute('data-value')) : 999;
                                const finValB = finCellB ? parseFloat(finCellB.getAttribute('data-value')) : 999;

                                // オッズが0.0の場合も除外扱い
                                const oddsValueA = parseFloat(cellA.getAttribute('data-value'));
                                const oddsValueB = parseFloat(cellB.getAttribute('data-value'));

                                const isWithdrawnA = (!isNaN(finValA) && finValA < 0) || (!isNaN(oddsValueA) &&
                                    oddsValueA === 0.0);
                                const isWithdrawnB = (!isNaN(finValB) && finValB < 0) || (!isNaN(oddsValueB) &&
                                    oddsValueB === 0.0);

                                if (isWithdrawnA !== isWithdrawnB) {
                                    return isWithdrawnA ? 1 : -1; // 取消等とオッズ0.0は常に下
                                }
                            }

                            // BB指数の特別処理：bb_point_rankでソート
                            if (column === 'bb-point') {
                                let valueA = parseFloat(cellA.getAttribute('data-value'));
                                let valueB = parseFloat(cellB.getAttribute('data-value'));

                                if (isNaN(valueA)) valueA = 9999;
                                if (isNaN(valueB)) valueB = 9999;

                                // directionに応じてソート方向を変更
                                return direction === 'asc' ? (valueA - valueB) : (valueB - valueA);
                            }

                            let valueA = parseFloat(cellA.getAttribute('data-value'));
                            let valueB = parseFloat(cellB.getAttribute('data-value'));

                            if (column === 'finish-position') {
                                // 特別値（未設定=999 or NaN、負値=取消等）を常に最下位へ
                                const isSpecialA = isNaN(valueA) || valueA === 999 || valueA < 0;
                                const isSpecialB = isNaN(valueB) || valueB === 999 || valueB < 0;
                                if (isSpecialA !== isSpecialB) {
                                    return isSpecialA ? 1 : -1; // specialは後ろ
                                }

                                // 100番台は100を引いて実着順として扱う
                                if (!isNaN(valueA) && valueA >= 100) valueA = valueA - 100;
                                if (!isNaN(valueB) && valueB >= 100) valueB = valueB - 100;

                                // 通常の比較（昇順/降順）
                                if (isNaN(valueA)) valueA = 9999;
                                if (isNaN(valueB)) valueB = 9999;
                            } else {
                                if (isNaN(valueA)) valueA = 0;
                                if (isNaN(valueB)) valueB = 0;
                            }

                            return direction === 'asc' ? (valueA - valueB) : (valueB - valueA);
                        });

                        // DOM更新
                        while (horseTbody.firstChild) {
                            horseTbody.removeChild(horseTbody.firstChild);
                        }
                        sortedRows.forEach(row => horseTbody.appendChild(row));
                    }

                    // ヘッダークリックでのソート切替
                    horseHeaders.forEach(header => {
                        header.addEventListener('click', function() {
                            const sortKey = this.getAttribute('data-sort');
                            const sortIcon = this.querySelector('.sort-icon');
                            if (!sortKey || !sortIcon) return;

                            // 既に同じカラムでソートされていれば、方向を逆にする
                            if (horseCurrentSort.column === sortKey) {
                                horseCurrentSort.direction = horseCurrentSort.direction === 'asc' ? 'desc' : 'asc';
                            } else {
                                horseCurrentSort.column = sortKey;
                                horseCurrentSort.direction = 'asc';

                                // 他のソートアイコンをリセット
                                horseTable.querySelectorAll('.sort-icon').forEach(icon => {
                                    icon.classList.remove('asc', 'desc');
                                });
                            }

                            // ソートアイコンの更新
                            sortIcon.classList.remove('asc', 'desc');
                            sortIcon.classList.add(horseCurrentSort.direction);

                            // ソート実行
                            sortHorseTable(sortKey, horseCurrentSort.direction);
                        });
                    });
                }

                                    break;

                                    cellA = a.querySelector('td:nth-child(5)');
                                    cellB = b.querySelector('td:nth-child(5)');
                                    break;
                            }

                            if (!cellA || !cellB) return 0;

                            let valueA = cellA.getAttribute('data-value');
                            let valueB = cellB.getAttribute('data-value');

                            valueA = parseFloat(valueA);

                            if (isNaN(valueA)) valueA = 0;
                            if (isNaN(valueB)) valueB = 0;

                                valueB - valueA;
                        });

                        while (frameTbody.firstChild) {
                            frameTbody.removeChild(frameTbody.firstChild);
                        }

                        sortedRows.forEach(row => {
                            frameTbody.appendChild(row);
                        });
                    }

                    // 設定をCookieに保存する関数
                    function saveColumnSettingsToCookie() {
                        const cookieName = 'column_settings';
                        const settings = {};

                        // すべてのチェックボックスの状態を取得
                        const columnCheckboxes = document.querySelectorAll('#columnsModal input[type="checkbox"]');
                        columnCheckboxes.forEach(checkbox => {
                            const columnName = checkbox.getAttribute('data-column');
                            settings[columnName] = checkbox.checked;
                        });

                        // Cookieに保存（30日間）
                        setCookie(cookieName, JSON.stringify(settings), 30);
                    }

                    // Cookieから設定を読み込む関数
                    function loadColumnSettingsFromCookie() {
                        const cookieName = 'column_settings';
                        const cookieValue = getCookie(cookieName);

                        if (cookieValue) {
                            try {
                                const settings = JSON.parse(cookieValue);

                                // チェックボックスの状態を設定
                                const columnCheckboxes = document.querySelectorAll(
                                    '#columnsModal input[type="checkbox"]');
                                columnCheckboxes.forEach(checkbox => {
                                    const columnName = checkbox.getAttribute('data-column');
                                    if (typeof settings[columnName] !== 'undefined') {
                                        checkbox.checked = settings[columnName];
                                        // カラムの表示・非表示を設定
                                        toggleColumnVisibility(columnName, settings[columnName]);
                                    }
                                });
                            } catch (e) {
                                console.error('Cookieの解析に失敗しました:', e);
                            }
                        }
                    }
                }

                document.addEventListener('DOMContentLoaded', function() {
                    // スクロール可能なコンテナ（テーブルラッパー）を取得
                    var tableContainer = document.querySelector('.table-responsive');
                    // 馬名カラムのヘッダー（馬名・騎手名・厩舎名）を取得
                    var thirdHeader = document.querySelector('#race-horses-table thead tr th.horse-info-col');
                    // tbody 内の馬名カラムの各セルをすべて取得
                    var thirdCells = document.querySelectorAll(
                        '#race-horses-table tbody tr td.horse-info-col');

                    // 要素が存在しない場合は処理をスキップ（障害レースやデータがない場合）
                    if (!tableContainer || !thirdHeader || !thirdCells || thirdCells.length === 0) {
                        return;
                    }

                    // スクロールの閾値をデバイス幅に応じて調整
                    var threshold;

                    function updateThreshold() {
                        var windowWidth = window.innerWidth;
                        threshold = 1;
                    }

                    // 初期閾値を設定
                    updateThreshold();

                    // 画面サイズ変更時にも閾値を更新
                    window.addEventListener('resize', updateThreshold);

                    // 切り替え時にアニメーションするためのクラス
                    var isVerticalMode = false;
                    var isTransitioning = false;

                    // 元々のスタイル情報を取得（ヘッダーの幅とフォントサイズ）
                    var originalHeaderWidth = window.getComputedStyle(thirdHeader).width;
                    var originalHeaderFontSize = window.getComputedStyle(thirdHeader).fontSize;

                    // 各セルの元の幅・フォントサイズ、さらに内部の <span> の display 状態も記録
                    var originalCellStyles = [];
                    var originalContents = [];
                    thirdCells.forEach(function(cell) {
                        var computed = window.getComputedStyle(cell);
                        // セル内の <span> があればその display 状態も取得
                        var span = cell.querySelector('span');
                        var originalSpanDisplay = span ? window.getComputedStyle(span).display :
                            null;
                        originalCellStyles.push({
                            fontSize: computed.fontSize,
                            width: computed.width,
                            spanDisplay: originalSpanDisplay
                        });

                        // オリジナルのHTML内容を保存
                        originalContents.push(cell.innerHTML);
                    });

                    // スクロールイベントリスナー
                    tableContainer.addEventListener('scroll', function() {
                        // 現在のスクロール位置
                        var scrollLeft = this.scrollLeft;

                        // スクロール位置が閾値を超えているかどうか
                        var shouldBeVertical = scrollLeft >= threshold;

                        // 現在の状態とスクロール位置から判断した状態が異なる場合のみ処理を実行
                        if (shouldBeVertical !== isVerticalMode && !isTransitioning) {
                            isTransitioning = true;

                            if (shouldBeVertical) {
                                // 横表示から縦表示へ切り替え
                                switchToVerticalMode();
                            } else {
                                // 縦表示から横表示へ切り替え
                                switchToHorizontalMode();
                            }

                            // 状態を更新
                            isVerticalMode = shouldBeVertical;

                            // トランジション完了後にフラグをリセット
                            setTimeout(function() {
                                isTransitioning = false;
                            }, 100); // 300msの遅延（トランジション時間と同じにすると良い）
                        }

                        // 固定領域の幅を計算
                        updateFixedAreaBoundary();
                    });

                    // 縦表示モードに切り替える関数
                    function switchToVerticalMode() {
                        // 要素が存在しない場合は処理をスキップ
                        if (!thirdHeader || !thirdCells || thirdCells.length === 0) {
                            return;
                        }
                        // ヘッダーの変更
                        thirdHeader.style.minWidth = '30px';
                        thirdHeader.innerHTML = '馬<br>名'; // ヘッダーテキストを簡略化

                        // 各セルを縦書きに変更
                        thirdCells.forEach(function(cell, index) {

                            // 馬名を取得して縦書きで表示（最大6文字まで）
                            var horseName = cell.textContent.trim().split('\n')[0]; // 最初の行（馬名）を取得
                            if (horseName.length > 4) {
                                horseName = horseName.substring(0, 4); // 6文字に制限
                            }

                            // 縦書きにするためのコンテナを作成
                            cell.innerHTML =
                                '<div class="vertical-horse-name" style="font-size: 10px; margin: 0 auto; text-align: center;">' +
                                horseName + '</div>';
                        });
                    }

                    // 横表示モードに切り替える関数
                    function switchToHorizontalMode() {
                        // 要素が存在しない場合は処理をスキップ
                        if (!thirdHeader || !thirdCells || thirdCells.length === 0) {
                            return;
                        }
                        // ヘッダーを元に戻す
                        thirdHeader.style.width = originalHeaderWidth;
                        thirdHeader.style.fontSize = originalHeaderFontSize;
                        thirdHeader.innerHTML = '馬名<br>騎手名・厩舎名';

                        // 各セルを元に戻す
                        thirdCells.forEach(function(cell, index) {
                            cell.style.width = originalCellStyles[index].width;
                            cell.style.fontSize = originalCellStyles[index].fontSize;
                            cell.style.textAlign = 'start';
                            cell.style.padding = '';

                            // 元のHTMLに戻す
                            cell.innerHTML = originalContents[index];
                        });

                        /*
                        // 4列目以降のセルをすべて表示状態にする
                        var allCells = document.querySelectorAll('#race-horses-table thead th:nth-child(n+4), #race-horses-table tbody td:nth-child(n+4)');
                        allCells.forEach(function(cell) {
                            cell.style.visibility = 'visible';
                        });
                        */
                    }

                        // 馬名カラムの現在の幅を取得

                            window.getComputedStyle(finishPositionHeader).display !== 'none';

                        } else {
                            // 馬番カラム(30) + 馬印カラム(30) + 馬名カラム(可変)
                            stickyAreaWidth = 30 + 30 + currentThirdWidth;
                        }
                    }
                });

                // 新聞版レスポンシブスケール処理関数
                function initNewspaperScale() {
                    // 新聞版タブが表示されているかチェック
                    const newspaperTab = document.getElementById('newspaper-content');
                    if (!newspaperTab || newspaperTab.style.display === 'none') {
                        return;
                    }

                    const baseWidth = 800; // ベース幅
                    const newspaperWrapper = document.querySelector('#newspaper-content .newspaper-content-wrapper');
                    const scrollWrapper = document.querySelector('#newspaper-content .newspaper-scroll-wrapper');

                    if (!newspaperWrapper) {
                        console.warn('新聞コンテンツ要素が見つかりません');
                        return;
                    }

                    // スケール処理開始時は一旦非表示にする
                    newspaperWrapper.classList.remove('ready');



                    function updateScale(retryCount = 0) {
                        // Chrome開発者ツールでのデバイスシミュレーション対応
                        const screenWidth = document.documentElement.clientWidth || window.innerWidth;

                        // 要素の高さが0の場合は読み込み未完了として再試行
                        if (newspaperWrapper.offsetHeight === 0 && retryCount < 10) {
                            setTimeout(() => {
                                updateScale(retryCount + 1);
                            }, 300);
                            return;
                        }

                        // 最大試行回数に達した場合の緊急対策
                        if (newspaperWrapper.offsetHeight === 0 && retryCount >= 10) {
                            console.warn('要素の高さが0のまま最大試行回数に達しました。CSSフォールバックに頼ります。');

                            // フォールバック時も表示する（画面が真っ白になるのを防ぐ）
                            newspaperWrapper.classList.add('ready');
                            return; // CSS min-height に頼る
                        }
                        // Android判定
                        const userAgent = navigator.userAgent;
                        const isAndroid = /Android/i.test(userAgent);

                        // 800px以上の場合はスケール処理を無効にして通常表示
                        if (screenWidth >= 800) {
                            newspaperWrapper.style.transform = '';
                            if (scrollWrapper) {
                                scrollWrapper.style.height = '';
                            }

                            // 800px以上でも表示する
                            newspaperWrapper.classList.add('ready');

                            return;
                        }

                        const containerPadding = 20; // 左右の余白
                        const availableWidth = screenWidth - containerPadding;

                        // スケール比率を計算（最大1.0、最小0.3）
                        let scale = Math.min(availableWidth / baseWidth, 1.0);
                        scale = Math.max(scale, 0.3);

                        // transform: scaleを適用
                        newspaperWrapper.style.transform = `scale(${scale})`;
                        newspaperWrapper.style.transformOrigin = 'top left';

                        // Android用の高さ計算強化
                        let finalScaledHeight;
                        if (isAndroid) {
                            // Android: 複数の高さ値を取得して最大値を使用
                            setTimeout(() => {
                                const heightOptions = [
                                    newspaperWrapper.offsetHeight,
                                    newspaperWrapper.scrollHeight,
                                    newspaperWrapper.getBoundingClientRect().height
                                ];
                                const maxHeight = Math.max(...heightOptions);
                                finalScaledHeight = maxHeight * scale;

                                if (scrollWrapper) {
                                    const androidPadding = 40; // Android用追加余白
                                    scrollWrapper.style.height = `${finalScaledHeight + androidPadding}px`;
                                    scrollWrapper.style.minHeight = `${finalScaledHeight + androidPadding}px`;
                                }

                                // Android用スケール処理完了後に表示
                                newspaperWrapper.classList.add('ready');
                            }, 200);
                        } else {
                            // 通常処理
                            finalScaledHeight = newspaperWrapper.offsetHeight * scale;
                            if (scrollWrapper) {
                                scrollWrapper.style.height = `${finalScaledHeight + 20}px`;
                            }

                            // 通常スケール処理完了後に表示
                            newspaperWrapper.classList.add('ready');
                        }
                    }



                    // 初期実行
                    updateScale();

                    // リサイズイベントでの再計算（デバウンス処理付き）
                    let resizeTimer;
                    window.addEventListener('resize', function() {
                        clearTimeout(resizeTimer);
                        resizeTimer = setTimeout(function() {
                            if (document.querySelector('#newspaper-content .newspaper-content-wrapper')) {
                                updateScale();
                            }
                        }, 100);
                    });

                    // オリエンテーション変更時の対応
                    window.addEventListener('orientationchange', function() {
                        setTimeout(function() {
                            if (document.querySelector('#newspaper-content .newspaper-content-wrapper')) {
                                updateScale();
                            }
                        }, 500);
                    });
                }
